\hypertarget{index_autotoc_md2}{}\doxysubsubsection{Naming}\label{index_autotoc_md2}

\begin{DoxyItemize}
\item Classes, class methods, functions, typedef or namespace names start with an upper case. The name \char`\"{}main\char`\"{} is the unavoidable exception.
\item Names for variables, class data members, function parameters, etc. must be all lower case. Start names for data member with a \char`\"{}m\+\_\+\char`\"{} prefix and for static data member names with a \char`\"{}s\+\_\+\char`\"{}. 
\end{DoxyItemize}
