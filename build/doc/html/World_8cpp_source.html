93  for (int i = 0; i < movedSubjectpath.size(); i++) {
94  float subject_Y0 = subject->getY();
95  float subject_Y1 = subject->getY() + subject->getHeight();

97  float movedSubject_y = movedSubjectpath[i].second;
98  if (goesUp)
99  movedSubject_y += movedSubject->getHeight();
100 
101  if (subject_Y0 <= movedSubject_y && movedSubject_y <= subject_Y1) {
102  float smallest_X0, smallest_X1, biggest_X0, biggest_X1;
103 
104  if (subject->getWidth() < movedSubject->getWidth()) {
105  smallest_X0 = subject->getX();
106  smallest_X1 = subject->getX() + subject->getWidth();
107 
108  biggest_X0 = movedSubject->getX();
109  biggest_X1 = movedSubject->getX() + movedSubject->getWidth();
110  }
111 
112  else {
113  biggest_X0 = subject->getX();
114  biggest_X1 = subject->getX() + subject->getWidth();
115  smallest_X0 = movedSubject->getX();
116  smallest_X1 = movedSubject->getX() + movedSubject->getWidth();
117  }
118 
119  if ((biggest_X0 <= smallest_X0 && smallest_X0 <= biggest_X1) ||
120  (biggest_X0 <= smallest_X1 && smallest_X1 <= biggest_X1)) {
121  return true;

129 template <typename T>
130 void World::getPointsBetweenFrames(vector<pair<float, float>>& left, const shared_ptr<T>& subject)
131 {
132 
133  ControllingPointers::control(subject, "World",
134  "getPointsBetweenFrames(vector<pair<float, float>> &left, vector<pair<float, "
135  "float>> &right, const shared_ptr<Player_L>& subject)");
136 
137  // Code gehaald uit project van vorig jaar voor het Project Computer Graphics.
138 
139  float x0 = subject->getXprev();
140  float x1 = subject->getX();
141  float y0 = subject->getYprev();
142  float y1 = subject->getY();
143 
144  float iterator = 0.001f;
145 
146  if (x0 == x1) {
147  for (float i = std::min(y0, y1); i <= std::max(y0, y1); i += iterator) {
148  left.emplace_back(x0, i);
149  }
150  } else if (y0 == y1) {
151  for (float i = std::min(x0, x1); i <= std::max(x0, x1); i += iterator) {
152  left.emplace_back(i, y0);
153  }
154  } else {
155  if (x0 > x1) {
156  std::swap(x0, x1);
157  std::swap(y0, y1);
158  }
159 
160  float m = (y1 - y0) / (x1 - x0);
161  if (-1.0 <= m && m <= 1.0) {
162  for (float i = 0; i <= (x1 - x0); i += iterator) {
163  left.emplace_back(x0 + i, (y0 + m * i));
164  }
165  } else if (m > 1.0) {
166  for (float i = 0; i <= (y1 - y0); i += iterator) {
167  left.emplace_back((x0 + (i / m)), y0 + i);
168  }
169  } else if (m < -1.0) {
170  for (float i = 0; i <= (y0 - y1); i += iterator) {
171  left.emplace_back((x0 - (i / m)), y0 - i);
172  }
173  }
174  }
